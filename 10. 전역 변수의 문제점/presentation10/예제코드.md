## 14.1 변수의 생명 주기
```javascript
function hi() {
    var hello = 'Hello World';
    console.log(hello); // -> Hello World
}
hi();   // 함수 호출
console.log(hello); // -> ReferenceError: hello is not defined
```
```javascript
var hello = 'Hello World';
function hi() {
    console.log(hello); // -> Hello World
    hello = 'Hi World';
}
hi();   // 함수 호출
console.log(hello); // -> Hi World
```
```javascript
function printHello() {
    var hello = 'Hello';
    console.log(hello); // Hello
}
printHello();
console.log(hello); // Reference error
```
```javascript
var x = 'global';

function foo() {
    console.log(x); // Undefined
    var x = 'local';
}

foo();
console.log(x); // global
```

## 14.3 전역 변수의 사용을 억제하는 방법
```javascript
function add(x, y) {
    return x + y;
}

console.log(add(3, 4));
```
```javascript
var x, y;
function add() {
    return x + y;
}

x = 3;
y = 4;
console.log(add());
```
```javascript
var ret = (function f() {
    var num = 2;
    return num;
})();

console.log(ret);   // 2
console.log(num);   // ReferenceError: num is not defined
```
```javascript
var MYAPP = {}; // 전역 네임스페이스 객체

MYAPP.person = {
  name: 'Lee',
  address: 'Seoul'
};

console.log(MYAPP); // person: {name: 'Lee', address: 'Seoul'} ...
console.log(MYAPP.person.name); // Lee
```
```javascript
var car = {
    position: 0
}
// 관련된 메서드가 묶여있지않음
var move = function(object) {
    object.position++;
}

console.log(car.position);  // 0
move(car);
console.log(car.position);  // 1
```
```javascript
var car = {
    position: 0,
    // 관련된 메서드가 묶여 있음
    move: function() {
        this.position++;
    }
}

console.log(car.position);  // 0
car.move();
console.log(car.position);  // 1
```
```javascript
var Counter = (function () {
  // private 변수
  var num = 0;

  // 외부로 공개할 데이터나 메서드를 프로퍼티로 추가한 객체를 반환한다.
  return {
    increase() {
      return ++num;
    },
    decrease() {
      return --num;
    }
  };
}());

// private 변수는 외부로 노출되지 않는다.
console.log(Counter.num); // undefined

console.log(Counter.increase()); // 1
console.log(Counter.increase()); // 2
console.log(Counter.decrease()); // 1
console.log(Counter.decrease()); // 0
```
```javascript
class Car {

    #mileage;

    constructor(meter) {    // 생성자
        this.#mileage = meter ?? 0;
    }

    get mileage() {         // getter
        return this.#mileage;
    }

    set mileage(meter) {    // setter
        this.#mileage = meter;
    }
}

var car = new Car();        // 생성자 호출
console.log(car.mileage);   // 0, getter 호출
car.mileage = 58038;        // setter 호출
console.log(car.mileage);   // 58038, getter 호출
// console.log(car.#mileage);  // SyntaxError: Private field '#mileage' must be declared in an enclosing class
```
